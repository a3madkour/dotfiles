** Smart-parens
Adapted from [[https://github.com/doomemacs/doomemacs/blob/master/modules/config/default/config.el][Doom Emacs's smartparens config]] (=modules/config/default/config.el=); the =(dolist (brace ...))= block and =:unless= rationale are verbatim.
#+begin_src emacs-lisp
(use-package smartparens
  :diminish smartparens-mode
  :defer 1
  :config
  ;; Load default smartparens rules for various languages
  (require 'smartparens-config)
  (setq sp-max-prefix-length 25)
  (setq sp-max-pair-length 4)
  (setq sp-highlight-pair-overlay nil
		sp-highlight-wrap-overlay nil
		sp-highlight-wrap-tag-overlay nil)

  (with-eval-after-load 'evil
	(setq sp-show-pair-from-inside t)
	(setq sp-cancel-autoskip-on-backward-movement nil)
	(setq sp-pair-overlay-keymap (make-sparse-keymap)))

  (let ((unless-list '(sp-point-before-word-p
					   sp-point-after-word-p
					   sp-point-before-same-p)))
	(sp-pair "'"  nil :unless unless-list)
	(sp-pair "\"" nil :unless unless-list))

  ;; In lisps ( should open a new form if before another parenthesis
  (sp-local-pair sp-lisp-modes "(" ")" :unless '(:rem sp-point-before-same-p))

  ;; Don't do square-bracket space-expansion where it doesn't make sense to
  (sp-local-pair '(emacs-lisp-mode org-mode markdown-mode gfm-mode)
				 "[" nil :post-handlers '(:rem ("| " "SPC")))


  (dolist (brace '("(" "{" "["))
	(sp-pair brace nil
			 :post-handlers '(("||\n[i]" "RET") ("| " "SPC"))
			 ;; Don't autopair opening braces if before a word character or
			 ;; other opening brace. The rationale: it interferes with manual
			 ;; balancing of braces, and is odd form to have s-exps with no
			 ;; whitespace in between, e.g. ()()(). Insert whitespace if
			 ;; genuinely want to start a new form in the middle of a word.
			 :unless '(sp-point-before-word-p sp-point-before-same-p)))
  (smartparens-global-mode t))

;;;###autoload
(defun +default/insert-file-path (arg)
  "Insert the file name (absolute path if prefix ARG).
	If `buffer-file-name' isn't set, uses `default-directory'."
  (interactive "P")
  (let ((path (or buffer-file-name default-directory)))
	(insert
	 (if arg
		 (abbreviate-file-name path)
	   (file-name-nondirectory path)))))
;;;###autoload
(defun +default/insert-all-path (arg)
  (interactive "P")
  (+default/insert-file-path 1))

  "Filter 'recentf-list' to show only files within the current project"
  (interactive)
  (let* ((proj (project-current t))
  		 (root (project-root proj))
  		 (recent-files (seq-filter (lambda(f) (string-prefix-p root f))

  (interactive)
  (let* ((project-root (project-root (project-current t)))
         (venv-path (expand-file-name ".venv" project-root))
         (python-path (expand-file-name
                       (if (eq system-type 'windows-nt)
                           "Scripts/python.exe"
                         "bin/python")
                       venv-path)))
    (if (file-exists-p python-path)
        (progn
          ;; Set Python interpreter path
          (setq python-shell-interpreter python-path)

          ;; Update exec-path to include the venv's bin directory

                                  (remove venv-bin-dir exec-path))))

          ;; Update PATH environment variable
          (setenv "PATH" (concat (file-name-directory python-path)
                                 path-separator
                                 (getenv "PATH")))

          ;; Update VIRTUAL_ENV environment variable
          (setenv "VIRTUAL_ENV" venv-path)

          ;; Remove PYTHONHOME if it exists
          (setenv "PYTHONHOME" nil)

          (message "Activated UV Python environment at %s" venv-path))
      (error "No UV Python environment found in %s" project-root))))

(defun a3madkour/citar-org-format-note(key entry)
  "Format a note from KEY and ENTRY."
  (let* ((template (citar--get-template 'note))
		 (note-meta (when template
					  (citar-format--entry template entry)))
		 (template-path (citar--get-template 'note-file))
		 (note-path (when template-path
					  (citar-format--entry template-path entry)))
		 (note-path-clean (when note-path
							(car (seq-filter (lambda (p) (string-match-p "\\.pdf$" (string-trim p)))
											 (split-string note-path ";")))))
		 (note-tags (cdr (citar-get-field-with-value '("keywords") key)))
		 (formatted-tags (when note-tags
						   (mapconcat (lambda (tag) (concat ":" (replace-regexp-in-string "\\s-+" "" (string-trim tag))))
									  (split-string note-tags "[,;]+")
									  "")))
		 (filepath (expand-file-name
					(concat key ".org")
					(car citar-notes-paths)))
		 (buffer (find-file filepath)))
	(with-current-buffer buffer
	  ;; This just overrides other template insertion.
	  (erase-buffer)
	  (citar-org-roam-make-preamble key)
	  ;; hacky way to insert NOTER_DOCUMENT
	  (goto-char (point-min))
	  (when (search-forward ":END:" nil t)
		(beginning-of-line)
		(insert ":NOTER_DOCUMENT: ")
		(when note-path-clean
		  (insert (concat "~/" (replace-regexp-in-string ".*Sync/" "Sync/" note-path-clean))))
		(insert "\n")	
		(insert ":DATE_CREATED: ")
		(insert (format-time-string "[%Y-%m-%d %a]"))
		(insert "\n")
		(insert ":LAST_MODIFIED: ")
		(insert (format-time-string "[%Y-%m-%d %a]"))
		(insert "\n"))	
	  )
	(goto-char (point-max))
	(insert "#+title: ")
	(when template (insert note-meta))
	(insert "\n#+filetags: :Ref")
	(when formatted-tags (insert formatted-tags))
	(insert ":\n")
	(insert "\n\n|\n\n#+print_bibliography:")
	(search-backward "|")
	(delete-char 1)
	(when (fboundp 'evil-insert)
	  (evil-insert 1))))

(defun a3madkour/citar-add-to-reading-queue ()
  "Select one or more citar references and add them to queue.org."
  (interactive)
  (require 'citar)
  (let* ((count (a3madkour/reading-queue-active-count))
         (limit a3madkour/reading-queue-wip-limit))
    (when (>= count limit)
      (user-error "Reading queue WIP limit reached (%d/%d active). Finish something first"
                  count limit))
    (let* ((keys (citar-select-refs))
           (n (length keys))
           (room (- limit count)))
      (when (> n room)
        (user-error "Selected %d ref(s) but only %d slot(s) left under WIP limit (%d/%d active)"
                    n room count limit))
      (with-current-buffer (find-file-noselect (a3madkour/org-file "queue.org"))
        (goto-char (point-max))
        (dolist (key keys)
          (let* ((entry (citar-get-entry key))
                 (title (alist-get "title" entry nil nil #'string=))
                 (author (alist-get "author" entry nil nil #'string=))
                 (year (when-let ((date (alist-get "date" entry nil nil #'string=)))
                         (substring date 0 4)))
                 (display (format "%s (%s)" (or title "Untitled") (or year "")))
                 (text (format "* TODO Read %s :active:toread:\n:PROPERTIES:\n:ENERGY: high\n:CITEKEY: %s\n:AUTHOR: %s\n:END:\n[cite:@%s]\n\n"
                               display key (or author "") key)))
            (insert text)))
        (save-buffer))
      (message "Added %d ref(s) to reading queue (%d/%d active)"
               n (+ count n) limit))))

         ((or (null src) (not (file-readable-p src)))
          (push key missing))
         (t
          (let ((dest (expand-file-name (file-name-nondirectory src) dest-dir)))
            (if (file-exists-p dest)
                (push (cons key (file-name-nondirectory src)) skipped)
              (copy-file src dest)
              (push (cons key (file-name-nondirectory src)) copied)))))))
    (with-current-buffer (get-buffer-create "*Supernote Sync*")

        (erase-buffer)

          (dolist (it (nreverse skipped))
            (insert (format "  %s  (%s)\n" (cdr it) (car it))))
          (insert "\n"))
        (when missing
          (insert "Missing PDF:\n")
          (dolist (k (nreverse missing))
            (insert (format "  %s\n" k)))))
      (special-mode))
    (display-buffer "*Supernote Sync*")
    (when (and a3madkour/supernote-reveal-outbox-after-sync copied)
      (pcase system-type
        ('darwin    (call-process "open" nil 0 nil dest-dir))
        ('gnu/linux (call-process "xdg-open" nil 0 nil dest-dir))))
    (message "Supernote outbox: %d copied, %d skipped, %d missing (drag from %s into OpenMTP)"
             (length copied) (length skipped) (length missing) dest-dir)))

                       '("-- Enter new tag --" "-- Done --")))

(defun a3madkour/toggle-active ()

  (interactive)

         (wip-entry (cl-find-if (lambda (entry)

                                a3madkour/wip-limits))

         (count-fn (when wip-entry (plist-get (cdr wip-entry) :count))))

	  (org-toggle-tag "active" 'on)

(defun a3madkour/create-project-notes ()
  "Create an org-roam note for the project at point and link it."
  (interactive)
  (unless (member "project" (org-get-tags))
    (user-error "Not on a project heading"))
  (let* ((title (org-get-heading t t t t))
         (note-title (read-string "Note title: " title ))
         (slug (org-roam-node-slug (org-roam-node-create :title note-title)))
         (dir (expand-file-name "projects" org-roam-directory))
         (filepath (expand-file-name (concat slug ".org") dir))
         (id nil)
         (has-context nil))
    (make-directory dir t)
    ;; Create the file if it doesn't exist, otherwise extract its ID
    (if (file-exists-p filepath)
        (with-current-buffer (find-file-noselect filepath)
		  (goto-char (point-min))
		  (when (re-search-forward "^:ID:\\s-+\\(.+\\)$" nil t)
            (setq id (match-string 1))))

        (insert (format ":PROPERTIES:\n:ID: %s\n:END:\n#+title: %s\n#+filetags: :project:\n\n* Context\n\n* Decisions\n\n* References\n"
                        id note-title))
        (save-buffer))
	  (org-roam-db-update-file filepath))
    ;; Add context link if not already present

	  (save-excursion
        (org-end-of-meta-data t)
        (let ((bound (save-excursion (org-end-of-subtree t t) (point))))
		  (setq has-context (re-search-forward "^Context: \\[\\[id:" bound t))))
	  (unless has-context
        (org-end-of-meta-data t)
        (insert (format "Context: [[id:%s][%s]]\n" id note-title))
        (save-buffer)))
    (find-file filepath)
    (message "Opened: %s" note-title)))

(defvar a3madkour--inbox-to-roam-marker nil
  "Marker to the inbox subtree being sent to org-roam, removed on capture finalize.")

(defun a3madkour--inbox-to-roam-finalize ()
  "Hook: cut the pending inbox subtree if the org-roam capture finalized successfully."
  (unwind-protect
      (when (and (not org-note-abort)
                 (markerp a3madkour--inbox-to-roam-marker)
                 (marker-buffer a3madkour--inbox-to-roam-marker))
        (with-current-buffer (marker-buffer a3madkour--inbox-to-roam-marker)
          (save-excursion
            (goto-char a3madkour--inbox-to-roam-marker)
            (org-cut-subtree))))
    (when (markerp a3madkour--inbox-to-roam-marker)
      (set-marker a3madkour--inbox-to-roam-marker nil))
    (setq a3madkour--inbox-to-roam-marker nil)
    (remove-hook 'org-capture-after-finalize-hook
                 #'a3madkour--inbox-to-roam-finalize)))

(defun a3madkour/inbox-to-org-roam ()
  "Send the current inbox item to org-roam as a Zettelkasten seed.
Preserves the inbox item's body as initial content of the new node.
The inbox subtree is removed only after the capture finalizes successfully;
on abort, the inbox item stays where it was."
  (org-back-to-heading t)
  (let* ((title (org-get-heading t t t t))
         (body (save-excursion
                 (org-end-of-meta-data t)
                 (string-trim
                  (buffer-substring-no-properties
                   (point)
                   (org-entry-end-position))))))
    (setq a3madkour--inbox-to-roam-marker (copy-marker (point) t))
    (add-hook 'org-capture-after-finalize-hook
              #'a3madkour--inbox-to-roam-finalize)
    (org-roam-capture-
     :node (org-roam-node-create :title title)
     :templates `(("d" "default" plain
                   ,(if (string-empty-p body)
                        "%?"
                      (concat body "\n\n%?"))
                   :target (file+head
                            "${slug}.org"
                            "#+title: ${title}\n#+filetags: :TODO:\n")
                   :unnarrowed t)))))

(defun a3madkour/-build-habit-repeater (frequency flexibility)
  "Compute the org-mode repeater string for a habit.
FREQUENCY is one of \"daily\", \"every 2 days\", \"every 3 days\",
\"weekly\", \"biweekly\", \"monthly\". FLEXIBILITY (a string) appends a
`/Nd' tolerance window but only for the daily/every-N-days frequencies."
  (let ((base (cond
               ((string= frequency "daily") ".+1d")
               ((string= frequency "every 2 days") ".+2d")
               ((string= frequency "every 3 days") ".+3d")
               ((string= frequency "weekly") ".+1w")
               ((string= frequency "biweekly") ".+2w")
               ((string= frequency "monthly") ".+1m"))))
    (if (and flexibility
             (not (string-empty-p flexibility))
             (member frequency '("daily" "every 2 days" "every 3 days")))
        (format "%s/%sd" base (string-to-number flexibility))
      base)))

(defun a3madkour/inbox-to-habit ()
  "Process the current inbox item as a recurring habit."
  (let* ((frequency (completing-read "How often? "

                                       "weekly" "biweekly" "monthly")
                                     nil t))
         (flexibility (when (member frequency '("daily" "every 2 days" "every 3 days"))
                        (read-string "Max days before overdue (blank for same): ")))
         (full-repeater (a3madkour/-build-habit-repeater frequency flexibility))
         (marker (a3madkour/get-refile-marker (a3madkour/org-file "actions.org") "Habits")))
    (org-set-property "STYLE" "habit")
    (org-schedule nil (format "<%s %s>"

                              full-repeater))
    (org-refile nil nil (list "Habits" (a3madkour/org-file "actions.org") nil marker))
    (set-marker marker nil))
  (message "Sent to habits"))

(defun a3madkour/process-inbox-item ()
  "Process the current inbox item: classify, set properties, and refile."
  (interactive)
  (unless (org-entry-get nil "TODO")
    (user-error "Not on a TODO heading"))
  (let ((destination (completing-read "Where does this go? "
                                      '("action" "project" "someday" "org-roam" "delete")
                                      nil t)))
    (cond
     ((string= destination "delete")
      (a3madkour/inbox-delete))
     ((string= destination "org-roam")
      (a3madkour/inbox-to-org-roam))
     ((string= destination "someday")
      (a3madkour/inbox-to-someday))
     ((string= destination "action")
      (if (yes-or-no-p "Is this a recurring habit? ")
          (a3madkour/inbox-to-habit)
        (a3madkour/inbox-to-action)))
     ((string= destination "project")
      (let ((project-action (completing-read "Which project? "
                                             '("existing project" "create new project")
                                             nil t)))
        (if (string= project-action "create new project")
            (a3madkour/inbox-to-new-project)
          (a3madkour/inbox-to-existing-project))))
     (t (user-error "Unknown destination: %s" destination)))))

(defun a3madkour/org-roam-insert-heading-link ()
  "Insert a link to a specific heading inside an org-roam node.
Prompts for a roam node (via the usual `org-roam-node-read'),
then for one of that node's headings, and inserts a
`[[file:PATH::#custom-id][heading]]' link (falling back to
`::*Heading' if the heading has no CUSTOM_ID). The target
heading is NOT made into a roam node, so the graph stays
clean. Use this to cite a specific lecture/chapter inside a
per-source reference file."
  (interactive)
  (let* ((node (org-roam-node-read))
         (file (org-roam-node-file node))
         (headings
          (with-current-buffer (find-file-noselect file)

             (org-map-entries
              (lambda ()
                (let* ((heading   (org-get-heading t t t t))
                       (path      (org-get-outline-path t))
                       (display   (mapconcat #'identity path " / "))
                       (custom-id (org-entry-get nil "CUSTOM_ID"))
                       (anchor    (if custom-id
                                      (concat "#" custom-id)
                                    (concat "*" heading))))
                  (list display heading anchor)))
              nil 'file))))
         (choice  (completing-read "Heading: " (mapcar #'car headings) nil t))
         (entry   (assoc choice headings))
         (heading (nth 1 entry))
         (anchor  (nth 2 entry)))
    (insert (format "[[file:%s::%s][%s]]" file anchor heading))))

(defun a3madkour/org-roam-rename-file-to-title ()
  "Rename the current org-roam file to match its #+title, but
only if the filename doesn't already match. Does nothing if
the file is not in `org-roam-directory' or has no title."
  (when (and (org-roam-file-p)

		   (node (org-roam-node-at-point))
		   (slug (when node (org-roam-node-slug node)))
		   (current-file (buffer-file-name))
		   (current-dir (file-name-directory current-file))
		   (current-name (file-name-nondirectory current-file))
		   (new-name (when slug (concat slug ".org"))))
	  (when (and title slug new-name
                 (not (string-equal current-name new-name)))
        (let ((new-path (expand-file-name new-name current-dir)))
		  (if (file-exists-p new-path)
			  (message "Org-roam: cannot rename, %s already exists" new-name)
            (rename-file current-file new-path)
            (set-visited-file-name new-path t t)
            (message "Org-roam: renamed to %s" new-name)))))))

      (kbd "RET") #'comint-send-input))

  (add-hook 'diff-mode-hook
            (lambda ()
              (when (string-match-p "\\*agent-shell-diff\\*" (buffer-name))
                (evil-emacs-state))))

  (add-hook 'agent-shell-mode-hook
            (lambda ()
              (setq-local corfu-auto t)
              (setq-local corfu-auto-prefix 1)
              (setq-local corfu-auto-delay 0.1)
			  (setq-local completion-at-point-functions
                          '(agent-shell--command-completion-at-point
							agent-shell--file-completion-at-point))

  ;; ── Fix: symlinked project dirs not found by project.el ──
  (advice-add 'agent-shell--project-files :around
              (lambda (orig-fn &rest args)
                (let ((default-directory (file-truename default-directory)))
                  (apply orig-fn args))))

  ;; ── Fix: OpenCode sends oldString/newString/filePath instead of
  ;;    old_str/new_str/path that agent-shell expects ──
  (advice-add 'agent-shell--make-diff-info :around
              (lambda (orig-fn &rest args)
                (or (apply orig-fn args)
                    (when-let* ((acp-tool-call (plist-get args :acp-tool-call))

                                                              '(params toolCall toolCallId)))
